	NAT:
		interfaz: enp0s3

	OnlyHost:
		interfaz: enp0s8
		direccion: 192.168.56.2/24

	RedInterna2:
		interfaz: enp0s9
		direccion: 192.168.1.100/24

	Configuraciones:
		Para permitir el enrutamiento de paquetes:
			echo 1 > /proc/sys/net/ipv4/ip_forward
			En la ruta '/etc/sysctl.conf' descomentar la linea "#net.ipv4.ip_forward=1"
			iptables -t nat -A POSTROUTING -s 192.168.56.100/24 -o enp0s3 -j MASQUERADE
		Para configurar el servidor dhcp:
			En la ruta '/etc/default/isc-dhcp-server' se ha añadido la linea "INTERFACESv4="enp0s9""
			En la ruta '/etc/dhcp/dhcpd.conf' se ha añadido lo siguiente:
				subnet 192.168.1.0 netmask 255.255.255.0 {
        				range 192.168.1.101 192.168.1.254;
        				option subnet-mask 255.255.255.0;
        				option broadcast-address 192.168.1.255;
				        option routers 192.168.1.100;
				}
